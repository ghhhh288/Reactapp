"use client";

import KeyboardKey from "./KeyboardKey";
import styles from "./CallControls.module.css";

export default function CallControls({
  disabled,
  canHangUp,
  onStartCall,
  onEndCall,
  onClear,
}) {
  return (
    <div className={styles.wrapper}>
      <KeyboardKey
        label="Clear"
        variant="default"
        onPress={onClear}
        disabled={disabled}
      />
      <KeyboardKey
        label={canHangUp ? "End" : "Call"}
        variant={canHangUp ? "danger" : "primary"}
        onPress={canHangUp ? onEndCall : onStartCall}
      />
    </div>
  );
}
